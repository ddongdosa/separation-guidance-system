    <script>
        // 현재 시간 업데이트
        function updateTime() {
            const now = new Date();
            const timeString = now.toLocaleString('ko-KR');
            document.getElementById('currentTime').textContent = timeString;
        }

        // 페이지 로드 시 시간 설정 및 매초 업데이트
        updateTime();
        setInterval(updateTime, 1000);

        // 페이지 로드 시 저장된 설정 불러오기
        window.addEventListener('load', function() {
            loadSettings();
        });

        // 설정 패널 토글
        function toggleSettings() {
            const panel = document.getElementById('settingsPanel');
            panel.style.display = panel.style.display === 'none' || panel.style.display === '' ? 'block' : 'none';
        }

        // 설정 저장
        function saveSettings() {
            const vicePrincipalEmail = document.getElementById('vicePrincipalEmail').value;
            const ccEmail = document.getElementById('ccEmail').value;
            const schoolName = document.getElementById('schoolName').value;

            if (!vicePrincipalEmail) {
                showStatus('교감선생님 메일 주소를 입력해주세요.', 'error');
                return;
            }

            // 메일 형식 검증
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(vicePrincipalEmail)) {
                showStatus('올바른 메일 주소 형식을 입력해주세요.', 'error');
                return;
            }

            // 브라우저 저장소에 설정 저장
            localStorage.setItem('vicePrincipalEmail', vicePrincipalEmail);
            localStorage.setItem('ccEmail', ccEmail);
            localStorage.setItem('schoolName', schoolName);

            showStatus('설정이 저장되었습니다.', 'success');
            toggleSettings();
        }

        // 설정 불러오기
        function loadSettings() {
            const vicePrincipalEmail = localStorage.getItem('vicePrincipalEmail');
            const ccEmail = localStorage.getItem('ccEmail');
            const schoolName = localStorage.getItem('schoolName');

            if (vicePrincipalEmail) {
                document.getElementById('vicePrincipalEmail').value = vicePrincipalEmail;
            }
            if (ccEmail) {
                document.getElementById('ccEmail').value = ccEmail;
            }
            if (schoolName) {
                document.getElementById('schoolName').value = schoolName;
            }
        }

        // 전송 기록 패널 토글
        function toggleHistory() {
            const panel = document.getElementById('historyPanel');
            panel.style.display = panel.style.display === 'none' || panel.style.display === '' ? 'block' : 'none';
            
            if (panel.style.display === 'block') {
                displayHistory();
            }
        }

        // 전송 기록을 localStorage에 저장
        function saveRequestHistory(data) {
            let history = JSON.parse(localStorage.getItem('requestHistory') || '[]');
            history.push(data);
            // 최대 50개까지만 저장
            if (history.length > 50) {
                history = history.slice(-50);
            }
            localStorage.setItem('requestHistory', JSON.stringify(history));
        }

        // 전송 기록 표시
        function displayHistory() {
            const history = JSON.parse(localStorage.getItem('requestHistory') || '[]');
            const historyList = document.getElementById('historyList');
            
            if (history.length === 0) {
                historyList.innerHTML = '<p>아직 전송된 요청이 없습니다.</p>';
                return;
            }

            historyList.innerHTML = history.map((item, index) => `
                <div class="history-item">
                    <strong>${index + 1}.</strong> ${item.date} | ${item.classTime || '시간미정'} | ${item.studentClass} ${item.studentName} | ${item.urgencyLevel}
                    <br><small>교사: ${item.teacherName} | 상황: ${item.situation.substring(0, 50)}${item.situation.length > 50 ? '...' : ''}</small>
                </div>
            `).join('');
        }

        // 엑셀 다운로드 함수
        function downloadExcel() {
            const history = JSON.parse(localStorage.getItem('requestHistory') || '[]');
            
            if (history.length === 0) {
                showStatus('다운로드할 기록이 없습니다.', 'error');
                return;
            }

            // CSV 형식으로 데이터 구성
            const csvHeaders = ['번호', '요청일시', '교시', '학급', '학생명', '교사명', '긴급도', '상황설명'];
            const csvData = history.map((item, index) => [
                index + 1,
                item.date,
                item.classTime || '미정',
                item.studentClass,
                item.studentName,
                item.teacherName,
                item.urgencyLevel,
                `"${item.situation.replace(/"/g, '""')}"`  // CSV를 위한 따옴표 처리
            ]);

            // CSV 문자열 생성
            const csvContent = [csvHeaders, ...csvData]
                .map(row => row.join(','))
                .join('\n');

            // BOM 추가 (한글 깨짐 방지)
            const bom = '\uFEFF';
            const blob = new Blob([bom + csvContent], { type: 'text/csv;charset=utf-8;' });
            
            // 파일 다운로드
            const link = document.createElement('a');
            const url = URL.createObjectURL(blob);
            link.setAttribute('href', url);
            
            const now = new Date();
            const fileName = `분리지도_요청기록_${now.getFullYear()}${(now.getMonth()+1).toString().padStart(2,'0')}${now.getDate().toString().padStart(2,'0')}.csv`;
            link.setAttribute('download', fileName);
            link.style.visibility = 'hidden';
            
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
            
            showStatus('엑셀 파일이 다운로드되었습니다.', 'success');
        }

        // 기록 삭제
        function clearHistory() {
            if (confirm('모든 전송 기록을 삭제하시겠습니까?')) {
                localStorage.removeItem('requestHistory');
                displayHistory();
                showStatus('모든 기록이 삭제되었습니다.', 'success');
            }
        }

        // 설정 초기화
        function resetSettings() {
            if (confirm('모든 설정을 초기화하시겠습니까?')) {
                localStorage.removeItem('vicePrincipalEmail');
                localStorage.removeItem('ccEmail');
                localStorage.removeItem('schoolName');
                
                document.getElementById('vicePrincipalEmail').value = '';
                document.getElementById('ccEmail').value = '';
                document.getElementById('schoolName').value = '';
                
                showStatus('설정이 초기화되었습니다.', 'success');
            }
        }

        // 메일 미리보기 함수
        function previewEmail() {
            const studentClass = document.getElementById('studentClass').value;

            if (!studentClass) {
                showStatus('학급을 선택해주세요.', 'error');
                return;
            }

            const classTime = document.getElementById('classTime').value || '시간미정';
            const teacherName = document.getElementById('teacherName').value || '담임교사';

        // 요청 전송 함수
        function sendRequest() {
            const studentClass = document.getElementById('studentClass').value;

            // 학급만 필수로 체크
            if (!studentClass) {
                showStatus('학급을 선택해주세요.', 'error');
                return;
            }

            // 설정된 메일 주소 확인
            const vicePrincipalEmail = localStorage.getItem('vicePrincipalEmail');
            if (!vicePrincipalEmail) {
                showStatus('설정에서 교감선생님 메일 주소를 먼저 설정해주세요.', 'error');
                toggleSettings();
                return;
            }

            // 나머지 정보는 기본값 사용 (빈 값일 경우)
            const classTime = document.getElementById('classTime').value || '시간미정';
            const teacherName = document.getElementById('teacherName').value || '담임교사';
            const studentName = document.getElementById('studentName').value || '해당 학급 학생';
            const urgencyLevel = document.getElementById('urgencyLevel').value || '매우긴급';
            const situation = document.getElementById('situation').value || '긴급상황 발생으로 즉시 분리지도가 필요합니다.\n추가 정보는 추후 보고드리겠습니다.';

            const currentTime = new Date().toLocaleString('ko-KR');
            const urgencyEmoji = urgencyLevel.includes('매우긴급') ? '🔴' : urgencyLevel.includes('긴급') ? '🟡' : '🟢';
            const schoolName = localStorage.getItem('schoolName') || '○○초등학교';
            const ccEmail = localStorage.getItem('ccEmail');
            
            // 메일 제목과 내용 구성
            const subject = `[${urgencyEmoji} ${urgencyLevel}] ${schoolName} 분리지도 요청 - ${studentClass} ${studentName} (${classTime})`;
            const body = `교감선생님께,

${schoolName} ${teacherName}입니다.
다음 학생에 대한 분리지도를 요청드립니다.

■ 요청 정보
- 요청 교사: ${teacherName}
- 요청 시간: ${currentTime}
- 학교: ${schoolName}
- 발생 교시: ${classTime}
- 대상 학생: ${studentClass} ${studentName}
- 긴급도: ${urgencyLevel}

■ 상황 설명
${situation}

■ 조치 요청사항
해당 학생에 대한 분리지도를 요청드립니다.
가능한 시간에 교무실로 학생 분리 조치 부탁드립니다.

감사합니다.

※ 이 메일은 ${schoolName} 분리지도 요청 시스템을 통해 자동 발송되었습니다.`;

            // 전송 기록 저장
            const requestData = {
                date: currentTime,
                classTime: classTime,
                studentClass: studentClass,
                studentName: studentName,
                teacherName: teacherName,
                urgencyLevel: urgencyLevel,
                situation: situation,
                recipient: vicePrincipalEmail
            };
            saveRequestHistory(requestData);

            // 메일 클라이언트 열기 (mailto 링크 사용)
            let mailtoLink = `mailto:${vicePrincipalEmail}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
            
            // CC 메일이 설정되어 있으면 추가
            if (ccEmail && ccEmail.trim()) {
                mailtoLink += `&cc=${encodeURIComponent(ccEmail)}`;
            }
            
            try {
                window.location.href = mailtoLink;
                showStatus(`⚡ 긴급 요청이 처리되었습니다!\n${currentTime}에 ${studentClass} ${classTime} 분리지도 요청이 ${vicePrincipalEmail}로 전송되었습니다.\n\n🔴 추가 정보는 나중에 보충해서 알려주셔도 됩니다.`, 'success');
                
                // 폼 초기화
                document.getElementById('separationForm').reset();
                document.getElementById('emailPreview').style.display = 'none';
                
                // 긴급도를 다시 기본값으로 설정
                document.getElementById('urgencyLevel').value = '매우긴급';
                
            } catch (error) {
                showStatus('메일 전송 중 오류가 발생했습니다. 다시 시도해주세요.', 'error');
            }
        }

        // 상태 메시지 표시 함수
        function showStatus(message, type) {
            const statusDiv = document.getElementById('status');
            statusDiv.textContent = message;
            statusDiv.className = `status ${type}`;
            statusDiv.style.display = 'block';

            // 3초 후 메시지 숨기기
            setTimeout(() => {
                statusDiv.style.display = 'none';
            }, 5000);
        }
